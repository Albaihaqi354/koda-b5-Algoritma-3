HitungDiskonBelanja
DECLARE nominal, diskon, hargaAkhir: num
  BEGIN
    INPUT(nominal)
    If nominal > 1.000.000
	diskon <- 0.10 * nominal
    ELSEIF nominal >= 500.000
	diskon <- 0.5 * nominal
	ELSE
	diskon <- 0
    ENDIF
	hargaAkhir <- nominal - diskon
    OUTPUT(hargaAkhir)
  END
